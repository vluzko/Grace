pub enum Stmt {
    AssignmentStmt {
        name: Identifier,
        expression: Node<Expr>,
    },
    LetStmt {
        name: Identifier,
        type_annotation: Option<Type>,
        expression: Node<Expr>,
    },
    FunctionDecStmt {
        name: Identifier,
        args: Vec<(Identifier, Type)>,
        kwargs: Vec<(Identifier, Type, Node<Expr>)>,
        block: Node<Block>,
        return_type: Type,
    },
    StructDec {
        name: Identifier,
        fields: Vec<(Identifier, Type)>,
    },
    IfStmt {
        condition: Node<Expr>,
        block: Node<Block>,
        else_block: Option<Node<Block>>,
    },
    WhileStmt {
        condition: Node<Expr>,
        block: Node<Block>,
    },
    ReturnStmt(Node<Expr>),
    YieldStmt(Node<Expr>),
    BreakStmt,
    PassStmt,
    ContinueStmt,
}
